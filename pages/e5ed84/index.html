# 垂直居中

  1. 行内元素
#text{
  height: 150px;
  line-height: 150px;  /*与height等值*/
}
1
2
3
4
  1. 多行行内元素
#parent{
  height: 150px;
  line-height: 30px;  /*元素在页面呈现为5行,则line-height的值为height/5*/
}
1
2
3
4
  1. 单个块级元素
/* 定位实现 */
#parent{
  height: 150px;
  position: relative;  /*父相*/
}
#son{
  position: absolute;  /*子绝*/
  top: 50%;  /*父元素高度一半,这里等同于top:75px;*/
  transform: translateY(-50%);  /*自身高度一半,这里等同于margin-top:-25px;*/
  height: 50px;
}

/* flex 实现 */
#parent{
  display: flex;
  align-items: center;
}
